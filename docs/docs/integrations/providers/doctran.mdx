from doctran.document_transformers import DoctranQATransformer

# Initialize the transformer
transformer = DoctranQATransformer()

# Load a document
with open('document.txt', 'r') as f:
    document = f.read()

# Define some questions
questions = [
    "What is the main topic of the document?",
    "Who is the author of the document?",
    "What are the key arguments presented in the document?"
]

# Run the transformer on the document and questions
responses = transformer.transform(document, questions)

# Print the responses
for i, response in enumerate(responses):
    print(f"Question {i+1}: {questions[i]}\nAnswer: {response}\n")


from doctran.document_transformers import DoctranPropertyExtractor

# Initialize the transformer
extractor = DoctranPropertyExtractor()

# Load a document
with open('document.txt', 'r') as f:
    document = f.read()

# Define the properties to extract
properties = [
    'title',
    'author',
    'date'
]

# Run the transformer on the document and properties
data = extractor.transform(document, properties)

# Print the extracted data
print(data)


from doctran.document_transformers import DoctranTextTranslator

# Initialize the translator
translator = DoctranTextTranslator()

# Load a document
with open('document.txt', 'r') as f:
    document = f.read()

# Translate the document from English to Spanish
translated_document = translator.translate(document, 'en', 'es')

# Print the translated document
print(translated_document)
